Interaktivni Python Turtle kod

Klikni na roze, podvučene linije koda kako bi pogledao video objašnjenje za njih!

            import turtle

            # 1. Priprema ekrana i kornjače
            turtle.title("Moja kućica")
            turtle.shape("turtle")
            turtle.width(4)
            turtle.speed(3)

            # 2. PLAVA BAZA KUĆICE
            turtle.color("blue")
            for _ in range(4):
                turtle.forward(150) # za crtanje u nazad se koristi turtle.backward
                turtle.left(90)

            # 3. ŠIRI CRVENI KROV (Popunjen crvenom bojom)
            turtle.penup()
            turtle.left(90)
            turtle.forward(150)    
            turtle.left(90)
            turtle.forward(25)     # Pomeramo se ulevo da krov bude širi (-25,150)
            turtle.right(180)      # Okrećemo kornjaču udesno (setheading(0))

            turtle.color("red")
            turtle.begin_fill()
            turtle.goto(75, 225)   # Vrh krova (sredina)
            turtle.goto(175, 150)  # Desni kraj krova
            turtle.goto(-25, 150)  # Povratak na početak krova
            turtle.end_fill()

            # 4. PROZORI (Preko stamp-a)
            turtle.shape("square")
            turtle.color("lightblue")
            turtle.penup()

            # Levi prozor
            turtle.goto(40, 100)
            turtle.shapesize(1.5, 1.5)
            turtle.stamp()

            # Desni prozor
            turtle.goto(110, 100)
            turtle.stamp()

            # 5. VRATA (Preko stamp-a)
            turtle.color("brown")
            turtle.goto(75, 30) 
            turtle.shapesize(3, 2) 
            turtle.stamp()

            # 6. KORNJAČA U DVORIŠTU
            turtle.shape("turtle")
            turtle.color("green")
            turtle.shapesize(1.5)
            turtle.goto(75, -50) 
            turtle.setheading(90) # Gleda ponosno u kuću

            turtle.done()
